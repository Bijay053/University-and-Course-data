Backend — new endpoints:
typescriptrouter.get("/settings/academic-levels", async (_req, res) => {
  const rows = await db.query(`SELECT * FROM academic_level_options ORDER BY sort_order`);
  res.json(rows);
});

router.post("/settings/academic-levels", async (req, res) => {
  const { name } = req.body;
  const result = await db.query(
    `INSERT INTO academic_level_options (name) VALUES ($1) ON CONFLICT (name) DO NOTHING RETURNING *`,
    [name]
  );
  res.json(result.rows[0]);
});

router.delete("/settings/academic-levels/:id", async (req, res) => {
  await db.query(`DELETE FROM academic_level_options WHERE id = $1`, [req.params.id]);
  res.json({ success: true });
});
