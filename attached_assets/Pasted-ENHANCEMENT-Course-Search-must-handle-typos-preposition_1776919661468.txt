ENHANCEMENT: Course Search must handle typos, preposition variations, and partial matches
Currently the search is strict/literal:

"bachelor of nursing" → 1 result ✓
"bachelor in nursing" → 0 results ❌
"bachlor nursing" (typo) → 0 results ❌
"sidney" (typo of Sydney) → 0 results ❌

Users make these mistakes constantly. Search must be forgiving.

IMPLEMENT fuzzy full-text search with 3 layers:
Layer 1 — Preposition & stopword normalization (query side)
Before searching, normalize the query:
typescriptconst STOPWORDS = new Set([
  'of', 'in', 'the', 'a', 'an', 'for', 'at', 'to', 'and', 
  'studies', 'degree', 'program', 'programme', 'course'
]);

function normalizeQuery(q: string): string[] {
  return q
    .toLowerCase()
    .replace(/[^\w\s]/g, ' ')       // remove punctuation
    .split(/\s+/)
    .filter(w => w.length >= 2 && !STOPWORDS.has(w))
    .map(stemWord);  // optional: basic stemming
}

function stemWord(w: string): string {
  // Simple rules: nursing→nurs, bachelors→bachelor
  return w
    .replace(/ings?$/, '')   // nursing → nurs
    .replace(/ies$/, 'y')    // studies → study
    .replace(/s$/, '');      // bachelors → bachelor
}

// Example:
// "bachelor in nursing"    → ['bachelor', 'nurs']
// "bachelor of nursing"    → ['bachelor', 'nurs']
// "bachelors nurse"        → ['bachelor', 'nurs']
// All produce SAME token set — guaranteed match
Layer 2 — Trigram fuzzy match in PostgreSQL (handles typos)
Already have pg_trgm extension available. Add GIN trigram indexes:
sqlCREATE EXTENSION IF NOT EXISTS pg_trgm;

CREATE INDEX IF NOT EXISTS idx_courses_name_trgm 
  ON courses USING gin (LOWER(name) gin_trgm_ops);

CREATE INDEX IF NOT EXISTS idx_universities_name_trgm 
  ON universities USING gin (LOWER(name) gin_trgm_ops);

CREATE INDEX IF NOT EXISTS idx_universities_city_trgm 
  ON universities USING gin (LOWER(city) gin_trgm_ops);

CREATE INDEX IF NOT EXISTS idx_courses_location_trgm
  ON courses USING gin (LOWER(course_location) gin_trgm_ops);
Search query with trigram similarity:
sql-- "bachlor nursing" (typo) will match "Bachelor of Nursing" with similarity > 0.3
SELECT *,
  GREATEST(
    similarity(LOWER(course_name), LOWER($1)),
    word_similarity(LOWER($1), LOWER(course_name))
  ) AS name_score
FROM course_search_view
WHERE 
  LOWER(course_name) % LOWER($1)  -- trigram operator: "similar enough"
  OR LOWER(university_name) % LOWER($1)
ORDER BY name_score DESC
LIMIT 20;
The % operator returns rows where similarity > pg_trgm.similarity_threshold (default 0.3 — tune to 0.25 for more forgiving).

Full-text search query:
sql-- to_tsquery automatically stems: 'nursing' and 'nurse' match same lexeme
SELECT *, ts_rank(search_tsv, query) AS fts_score
FROM course_search_view, plainto_tsquery('english', $1) query
WHERE search_tsv @@ query
ORDER BY fts_score DESC;

COMBINED SEARCH STRATEGY (the magic):
Run both trigram AND tsvector in parallel, combine scores:
typescriptasync function fuzzySearchCourses(userQuery: string, filters: Filters) {
  const query = userQuery.trim();
  
  const sql = `
    WITH fts_results AS (
      SELECT id, ts_rank(search_tsv, plainto_tsquery('english', $1)) AS score, 'fts' as method
      FROM course_search_view
      WHERE search_tsv @@ plainto_tsquery('english', $1)
    ),
    trigram_results AS (
      SELECT id, GREATEST(
        similarity(LOWER(course_name), LOWER($1)),
        similarity(LOWER(university_name), LOWER($1)),
        similarity(LOWER(COALESCE(course_location, '')), LOWER($1))
      ) AS score, 'trigram' as method
      FROM course_search_view
      WHERE 
        LOWER(course_name) % LOWER($1) 
        OR LOWER(university_name) % LOWER($1)
        OR LOWER(COALESCE(course_location, '')) % LOWER($1)
    ),
    combined AS (
      SELECT id, MAX(score) as score
      FROM (
        SELECT * FROM fts_results
        UNION ALL
        SELECT * FROM trigram_results
      ) merged
      GROUP BY id
    )
    SELECT v.*, c.score
    FROM combined c
    JOIN course_search_view v ON v.id = c.id
    ORDER BY c.score DESC
    LIMIT 20;
  `;

  return db.execute(sql, [query]);
}
This handles:

"bachelor of nursing" → FTS matches (stopwords ignored)
"bachelor in nursing" → FTS matches (stopwords ignored, same stem)
"bachlor nursing" → trigram matches (similarity > 0.3)
"nursing degree" → FTS matches via nursing tsvector
"nurs" → trigram matches nursing
"sidney" → trigram matches Sydney (similarity ~0.67)


LOCATION SEARCH (handles country + city + aliases):
Separate location search uses trigram + manual aliases:
typescriptconst CITY_ALIASES: Record<string, string[]> = {
  'sydney': ['sydney', 'syd', 'nsw'],
  'melbourne': ['melbourne', 'melb', 'vic'],
  'brisbane': ['brisbane', 'bris', 'qld'],
  'aussie': ['australia'],
  'oz': ['australia'],
};

function expandLocationQuery(loc: string): string[] {
  const key = loc.toLowerCase().trim();
  return CITY_ALIASES[key] ?? [loc];
}
