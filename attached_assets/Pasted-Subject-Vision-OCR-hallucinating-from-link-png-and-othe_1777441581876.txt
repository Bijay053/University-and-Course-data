    # social icons
    r"facebook", r"twitter", r"instagram", r"linkedin", r"youtube", r"tiktok",
    # UI affordances
    r"^link\.", r"^arrow\.", r"^external\.", r"^chain\.",

def _is_non_data_image(url: str) -> bool:
    filename = url.lower().split("/")[-1]
    return any(re.search(pat, filename) for pat in NON_DATA_IMAGE_PATTERNS)Apply this BEFORE Gemini Vision is called. We're paying API cost to hallucinate from icons.Fix 2 (URGENT): Validate OCR output looks like a real English requirements tableEven with filtering, edge cases will slip through. Add a post-OCR check that REJECTS results that don't look like a real table:pythondef _ocr_result_is_valid_english_table(ocr_text: str | None,
                                        returned_fields: dict) -> bool:
    """An OCR result is valid only if the OCR text looks like a real
    English requirements table — not just plausible numbers."""
    if not ocr_text:
        return False
    lower = ocr_text.lower()

    # The OCR-recovered TEXT must mention test names that match the
    # values being claimed. If we're claiming IELTS=6.5 and TOEFL=85,
    # the OCR text must contain "IELTS" AND "TOEFL".
    claimed_tests = []
    if returned_fields.get("ielts_overall"):
        claimed_tests.append("ielts")
    if returned_fields.get("toefl_overall"):
        claimed_tests.append("toefl")
    if returned_fields.get("pte_overall"):
        claimed_tests.append("pte")
    if returned_fields.get("cambridge_overall"):
        claimed_tests.append("cambridge")
    if returned_fields.get("duolingo_overall"):
        claimed_tests.append("duolingo")

    for test in claimed_tests:
        if test not in lower:
            return False  # claimed value, but test name not in OCR text → hallucination
    return TrueDiscard OCR results that fail this. No evidence row written.
