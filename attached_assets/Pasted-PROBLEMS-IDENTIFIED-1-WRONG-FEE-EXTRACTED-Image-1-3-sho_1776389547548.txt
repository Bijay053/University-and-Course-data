FIX 3: CORRECT FEE EXTRACTION FOR VIT
VIT shows fees in a specific format:
Fees: $48,000 ($3,000/unit)
Add VIT-specific fee extractor:
pythondef _extract_vit_fee(self, html: str) -> Dict:
    """
    VIT-specific fee extraction
    
    Pattern: "$48,000 ($3,000/unit)"
    or: "$51,000 ($2,125/unit) - 24 Units"
    """
    
    # Pattern 1: Full course fee with unit fee
    pattern1 = r'\$?([\d,]+)\s*\(\$?([\d,]+)/unit\)'
    match = re.search(pattern1, html)
    
    if match:
        full_course_fee = float(match.group(1).replace(',', ''))
        
        return {
            'international_fee': full_course_fee,
            'fee_term': 'Full Course',
            'fee_year': 2026
        }
    
    # Pattern 2: Just dollar amount
    pattern2 = r'Fees[:\s]*\$?([\d,]+)'
    match = re.search(pattern2, html)
    
    if match:
        fee = float(match.group(1).replace(',', ''))
        
        # Determine if this is full course or annual
        # Check nearby text
        context = html[max(0, match.start()-100):match.end()+100]
        
        if 'full course' in context.lower():
            fee_term = 'Full Course'
        elif 'per year' in context.lower() or 'annual' in context.lower():
            fee_term = 'Annual'
        else:
            # VIT typically shows full course fees
            fee_term = 'Full Course'
        
        return {
            'international_fee': fee,
            'fee_term': fee_term,
            'fee_year': 2026
        }
    
    return {}

pythonasync def discover_vit_courses(self, base_url: str) -> List[str]:
    """
    Discover all VIT courses from course list page
    """
    
    course_categories = ['bits', 'mits', 'mba', 'bbus', 'vocational', 'elicos']
    all_courses = []
    
    for category in course_categories:
        url = f'https://vit.edu.au/course-list?course_categories[0]={category}'
        
        print(f"[INFO] Fetching VIT {category.upper()} courses from: {url}")
        
        html = await self._fetch_html(url)
        soup = BeautifulSoup(html, 'html.parser')
        
        # Find all course links
        for link in soup.find_all('a', href=True):
            href = link.get('href')
            
            # VIT course URLs: /bits/..., /mits/..., /mba/...
            if any(f'/{cat}/' in href for cat in course_categories):
                full_url = urljoin('https://vit.edu.au', href)
                if full_url not in all_courses:
                    all_courses.append(full_url)
    
    print(f"[INFO] Discovered {len(all_courses)} VIT courses")
    
    return all_courses
