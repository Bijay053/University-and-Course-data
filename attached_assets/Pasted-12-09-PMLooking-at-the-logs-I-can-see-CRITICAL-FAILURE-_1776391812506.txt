IMPLEMENTATION: HYBRID FALLBACK
pythonasync def extract_course_with_fallback(url: str, university: University) -> Dict:
    """
    Try browser first, fallback to static HTML on failure
    """
    
    # Try browser automation
    try:
        print(f"[INFO] Trying browser for {url}")
        
        result = await extract_with_browser(url, university)
        
        # Check if successful
        if result.get('international_fee') and result.get('ielts_overall'):
            print(f"[SUCCESS] Browser extraction complete")
            return result
        else:
            raise Exception("Browser returned incomplete data")
    
    except Exception as e:
        print(f"[WARNING] Browser failed: {e}")
        print(f"[INFO] Falling back to static HTML")
        
        # Fallback to static HTML
        html = await fetch_html(url)
        result = extract_with_static_html(html, university)
        
        # Mark as incomplete
        result['extraction_method'] = 'static_html_fallback'
        result['data_completeness'] = 'partial'
        
        return result
