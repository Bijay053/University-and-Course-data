  SELECT sc.id AS scraped_course_id, 'ielts_overall' AS field_key, sc.ielts_overall::text AS col_val FROM scraped_courses sc WHERE sc.ielts_overall IS NOT NULL
  UNION ALL SELECT sc.id, 'pte_overall', sc.pte_overall::text FROM scraped_courses sc WHERE sc.pte_overall IS NOT NULL
  UNION ALL SELECT sc.id, 'toefl_overall', sc.toefl_overall::text FROM scraped_courses sc WHERE sc.toefl_overall IS NOT NULL
  UNION ALL SELECT sc.id, 'cambridge_overall', sc.cambridge_overall::text FROM scraped_courses sc WHERE sc.cambridge_overall IS NOT NULL

  UNION ALL SELECT sc.id, 'international_fee', sc.international_fee::text FROM scraped_courses sc WHERE sc.international_fee IS NOT NULL
),
ranked AS (
  SELECT sfe.id AS ev_id,
    ROW_NUMBER() OVER (PARTITION BY sfe.scraped_course_id, sfe.field_key ORDER BY sfe.confidence DESC NULLS LAST, sfe.id ASC) AS rn
  FROM scraped_field_evidence sfe
  JOIN course_cols cc ON cc.scraped_course_id = sfe.scraped_course_id AND cc.field_key = sfe.field_key
    AND ABS(COALESCE(sfe.normalized_value::numeric, -9999999) - COALESCE(cc.col_val::numeric, -9999998)) < 0.001
  WHERE sfe.decision_status != 'superseded'
)
UPDATE scraped_field_evidence sfe
SET decision_status = 'selected', selected = true
FROM ranked r
WHERE sfe.id = r.ev_id AND r.rn = 1 AND sfe.decision_status != 'selected';
