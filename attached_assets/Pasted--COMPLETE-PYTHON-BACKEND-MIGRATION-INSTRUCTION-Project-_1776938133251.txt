python"""Verify SQLAlchemy models match existing PostgreSQL schema."""
import asyncio
from sqlalchemy import inspect, text
from app.database import engine
from app.models import (
    University, Course, ScrapedCourse, Evidence,
    EnglishRequirement, Intake, Fee, ScrapeJob, BulkSession, User
)

async def verify():
    async with engine.begin() as conn:
        for model in [University, Course, ScrapedCourse, Evidence,
                      EnglishRequirement, Intake, Fee, ScrapeJob, BulkSession, User]:
            table_name = model.__tablename__
            result = await conn.execute(text(f"""
                SELECT column_name, data_type 
                FROM information_schema.columns 
                WHERE table_name = '{table_name}' 
                ORDER BY ordinal_position
            """))
            db_cols = {row[0]: row[1] for row in result}
            model_cols = {c.name: str(c.type) for c in inspect(model).columns}
            
            missing = set(db_cols.keys()) - set(model_cols.keys())
            extra = set(model_cols.keys()) - set(db_cols.keys())
            
            print(f"\n{table_name}:")
            print(f"  DB columns: {len(db_cols)}, Model columns: {len(model_cols)}")
            if missing: print(f"  ❌ Missing in model: {missing}")
            if extra: print(f"  ⚠️  Extra in model: {extra}")
            if not missing and not extra: print(f"  ✅ Match")

asyncio.run(verify())

from playwright.async_api import async_playwright, Browser, BrowserContext

class BrowserPool:
    def __init__(self, max_size: int = 5):
        self.max_size = max_size
        self._semaphore = asyncio.Semaphore(max_size)
        self._browser: Browser | None = None
        self._playwright = None
    
    async def start(self):
        self._playwright = await async_playwright().start()
        self._browser = await self._playwright.chromium.launch(headless=True)
    
    async def acquire_context(self) -> BrowserContext:
        await self._semaphore.acquire()
        return await self._browser.new_context()
    
    async def release(self, context: BrowserContext):
        await context.close()
        self._semaphore.release()
    
    async def stop(self):
        await self._browser.close()
        await self._playwright.stop()

        return response.text
backend-py/app/services/ai/budget.py:
pythonimport redis.asyncio as redis
from datetime import date

class BudgetTracker:
    def __init__(self, daily_limit: float):
        self.daily_limit = daily_limit
        self.redis = redis.from_url(settings.redis_url)
    
    async def check(self) -> bool:
        key = f"gemini:spend:{date.today().isoformat()}"
        spent = float(await self.redis.get(key) or 0)
        return spent < self.daily_limit
    
    async def record(self, estimated_cost: float):
        key = f"gemini:spend:{date.today().isoformat()}"
        await self.redis.incrbyfloat(key, estimated_cost)
        await self.redis.expire(key, 86400 * 2)  # 2 day TTL
