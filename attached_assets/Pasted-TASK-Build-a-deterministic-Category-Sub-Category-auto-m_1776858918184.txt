/**
 * Map a course name + degree level to a category and sub category.
 * Returns the highest-priority match. If nothing matches, returns null.
 */
export function mapCourseToCategory(courseName: string): { category: string; subCategory: string } | null {
  if (!courseName) return null;
  const normalized = courseName.trim();
  if (!normalized) return null;

  let bestMatch: { cat: string; sub: string; priority: number } | null = null;
  for (const rule of KEYWORD_RULES) {
    if (rule.pattern.test(normalized)) {
      if (!bestMatch || rule.priority > bestMatch.priority) {
        bestMatch = { cat: rule.cat, sub: rule.sub, priority: rule.priority };
      }
    }
  }

  if (bestMatch) {
    return { category: bestMatch.cat, subCategory: bestMatch.sub };
  }
  return null;
}

/**
 * Validate that a category/subCategory combination exists in the canonical taxonomy.
 * Useful for AI-returned values — if AI hallucinates, fall back to "Other".
 */
export function validateTaxonomy(category: string, subCategory: string): { category: string; subCategory: string } {
  const catEntry = COURSE_TAXONOMY.find(t => t.category.toLowerCase() === category.toLowerCase());
  if (!catEntry) {
    return { category: "Arts, Humanities & Social Sciences", subCategory: "Other" };
  }
  const subExists = catEntry.subCategories.find(s => s.toLowerCase() === subCategory.toLowerCase());
  return {
    category: catEntry.category,
    subCategory: subExists ?? "Other",
  };
}

/**
 * Map a free-text degree string to a canonical degree level.
 */
export function mapDegreeLevel(rawDegree: string, courseName: string): string {
  const text = `${rawDegree ?? ""} ${courseName ?? ""}`.toLowerCase();
  if (/\b(phd|doctorate|doctor of)\b/.test(text)) return "Doctor/Doctorate";
  if (/\b(dual|double) (degree|major)\b/.test(text)) return /master/.test(text) ? "Master Dual Degree" : "Bachelor Dual Degree";
  if (/\b(master|mba|mit|m\.?eng|llm|m\.?sc|m\.?a\.?)\b/.test(text)) return "Master";
  if (/\b(graduate certificate|graduate diploma|gdba|gcba|gdits|gcits)\b/.test(text)) return "Graduate Certificate & Diploma";
  if (/\b(bachelor|b\.?sc|b\.?a\.?|b\.?eng|llb|bbus|bits)\b/.test(text)) return "Bachelor";
  if (/\b(associate degree)\b/.test(text)) return "Associate Degree or Equivalent";
  if (/\bfoundation|pathway|preparation\b/.test(text)) return "Pathway to Undergraduate";
  if (/\b(certificate|diploma|advanced diploma)\b/.test(text)) return "Certificate & Diploma";
  if (/\b(english|elicos|esl|tesol)\b/.test(text)) return "English Language";
  return "Bachelor"; // safe default
}

    const validated = validateTaxonomy(course.aiCategory, course.aiSubCategory);
    course.category = validated.category;
    course.subCategory = validated.subCategory;
  } else {
    // Step 3: Last resort — never leave empty
    course.category = "Arts, Humanities & Social Sciences";
    course.subCategory = "Other";
  }

  // Same idea for degree level
  if (!course.degreeLevel) {
    course.degreeLevel = mapDegreeLevel(course.rawDegree ?? "", course.name);
  }

  addLog(job, "info", { message: `[CATEGORY] ${course.name} → ${course.category} / ${course.subCategory} (${detMap ? "deterministic" : course.aiCategory ? "ai+validated" : "default"})` });
}
