sudo -u postgres psql university_portal -c "
SELECT COUNT(*) AS suffix_count
FROM scraped_courses
WHERE university_id = 51 AND course_name LIKE '%| Acu Courses%';

-- Preview the cleaned names
SELECT id, course_name,
       trim(split_part(course_name, '|', 1)) AS cleaned_name
FROM scraped_courses
WHERE university_id = 51 AND course_name LIKE '%| Acu Courses%'
LIMIT 10;"

# Apply the name fix
sudo -u postgres psql university_portal -c "
UPDATE scraped_courses
SET course_name = trim(split_part(course_name, '|', 1))
WHERE university_id = 51 AND course_name LIKE '%| Acu Courses%'
RETURNING id, course_name;"
