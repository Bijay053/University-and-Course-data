2. Field Parsing Layer

Example:

function parseDuration(text) {
  if (!text) return null
  
  if (text.match(/(\d+)\s*year/)) return `${RegExp.$1} Year`
  if (text.match(/(\d+)\s*month/)) return `${Math.ceil(RegExp.$1/12)} Year`
  
  return null
}
function parseFee(text) {
  let match = text.match(/A?\$?[\d,]+/)
  return match ? match[0].replace(',', '') : null
}
function parseIELTS(text) {
  let match = text.match(/IELTS.*?(\d\.\d)/i)
  return match ? match[1] : null
}

function validateCourse(course) {
  let errors = []

  if (!course.fee) errors.push("Missing fee")
  if (!course.ielts) errors.push("Missing IELTS")
  if (course.duration > 6) errors.push("Invalid duration")

  return errors
}
