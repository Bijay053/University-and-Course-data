typescriptasync function runBulkQueue(universityIds: number[]) {
  const results: BulkResult[] = [];
  const PER_UNI_TIMEOUT_MS = 15 * 60 * 1000; // 15 min hard cap
  
  for (let i = 0; i < universityIds.length; i++) {
    const uniId = universityIds[i];
    const uni = await getUniversity(uniId);
    
    addLog({
      event: 'bulk_progress',
      message: `[BULK ${i+1}/${universityIds.length}] Starting: ${uni.name}`,
    });
    
    const startTime = Date.now();
    let result: BulkResult;
    
    try {
      // Hard timeout wrapper — if scrape hangs, we move on
      result = await Promise.race([
        scrapeOneUniversity(uniId),
        new Promise<BulkResult>((_, rej) => 
          setTimeout(() => rej(new Error(`bulk_timeout after ${PER_UNI_TIMEOUT_MS/60_000}min`)), PER_UNI_TIMEOUT_MS)
        ),
      ]);
    } catch (err: any) {
      // ALWAYS continue to next university, no matter what failed
      result = {
        universityId: uniId,
        status: 'failed',
        error: err.message.slice(0, 200),
        imported: 0,
        found: 0,
        durationMs: Date.now() - startTime,
      };
      addLog({
        event: 'bulk_error',
        message: `[BULK ${i+1}/${universityIds.length}] ${uni.name} FAILED: ${err.message} — moving to next`,
      });
    }
    
    results.push(result);
    
    // Force-release any locks between universities
    await releaseAllScrapeLocks();
    await killOrphanWorkers();
    await sleep(5000); // 5s cooldown
  }
  
  // Summary
  const succeeded = results.filter(r => r.status === 'completed');
  const failed = results.filter(r => r.status === 'failed');
  
  addLog({
    event: 'bulk_done',
    message: `[BULK-DONE] ${results.length} universities | ${succeeded.length} success | ${failed.length} failed`,
  });
  
  for (const r of results) {
    const icon = r.status === 'completed' ? '✓' : '✗';
    addLog({
      event: 'bulk_summary',
      message: `${icon} ${r.universityName} (${r.imported}/${r.found} in ${Math.round(r.durationMs/60000)}m) ${r.error || ''}`,
    });
  }
  
  return results;
}
