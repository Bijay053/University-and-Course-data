typescriptconst stagedCourses = await db.execute(sql`
  SELECT 
    sc.*,
    COALESCE((
      SELECT jsonb_agg(
        jsonb_build_object(
          'id', id,
          'fieldKey', field_key,
          'candidateValue', candidate_value,
          'normalizedValue', normalized_value,
          'sourceUrl', source_url,
          'pageType', page_type,
          'extractionMethod', extraction_method,
          'snippet', snippet,
          'confidence', confidence,
          'decisionScore', decision_score,
          'decisionStatus', decision_status,
          'validationStatus', validation_status,
          'selected', selected
        )
        ORDER BY field_key ASC, selected DESC, decision_score DESC NULLS LAST
      )
      FROM scraped_field_evidence
      WHERE scraped_course_id = sc.id
    ), '[]'::jsonb) as evidence
  FROM scraped_courses sc
  WHERE sc.scrape_job_id = ${jobId}
  ORDER BY sc.course_name ASC
`);
