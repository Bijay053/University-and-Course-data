CREATE MATERIALIZED VIEW public.course_search_view AS
 SELECT c.id,
    c.name AS course_name,
    c.category,
    c.sub_category,
    c.degree_level,
    c.duration,
    c.duration_term,
    c.study_mode,
    c.course_website,
    c.course_location,
    c.university_id,
    (setweight(to_tsvector('simple'::regconfig, COALESCE(c.name, ''::text)), 'A'::"char") || setweight(to_tsvector('simple'::regconfig, COALESCE(u.name, ''::text)), 'B'::"char")) AS name_tsv,
    ((((((setweight(to_tsvector('english'::regconfig, COALESCE(c.name, ''::text)), 'A'::"char") || setweight(to_tsvector('english'::regconfig, COALESCE(u.name, ''::text)), 'B'::"char")) || setweight(to_tsvector('english'::regconfig, COALESCE(c.category, ''::text)), 'C'::"char")) || setweight(to_tsvector('english'::regconfig, COALESCE(c.course_location, ''::text)), 'C'::"char")) || setweight(to_tsvector('english'::regconfig, COALESCE(u.city, ''::text)), 'C'::"char")) || setweight(to_tsvector('english'::regconfig, COALESCE(u.country, ''::text)), 'C'::"char")) || setweight(to_tsvector('english'::regconfig, COALESCE(c.sub_category, ''::text)), 'D'::"char")) AS search_tsv,
    u.name AS university_name,
    u.logo_url,
    u.city AS university_city,
    u.country AS university_country,
    u.website AS university_website,
    COALESCE(u.featured, false) AS featured,
    COALESCE(u.featured_priority, 0) AS featured_priority,
    ( SELECT fees.international_fee
           FROM public.fees
          WHERE (fees.course_id = c.id)
          ORDER BY fees.id DESC
         LIMIT 1) AS international_fee,
    ( SELECT fees.currency
           FROM public.fees
          WHERE (fees.course_id = c.id)
          ORDER BY fees.id DESC
         LIMIT 1) AS currency,
    ( SELECT fees.fee_term
           FROM public.fees
          WHERE (fees.course_id = c.id)
          ORDER BY fees.id DESC
         LIMIT 1) AS fee_term,
    NULL::real AS application_fee,
    ( SELECT array_agg(DISTINCT intakes.intake_month ORDER BY intakes.intake_month) AS array_agg
           FROM public.intakes
          WHERE ((intakes.course_id = c.id) AND (intakes.intake_month IS NOT NULL))) AS intakes,
    ( SELECT english_requirements.overall
           FROM public.english_requirements
          WHERE ((english_requirements.course_id = c.id) AND (english_requirements.test_type = 'IELTS'::text))
         LIMIT 1) AS ielts_overall,
    ( SELECT english_requirements.overall
           FROM public.english_requirements
          WHERE ((english_requirements.course_id = c.id) AND (english_requirements.test_type = 'PTE'::text))
         LIMIT 1) AS pte_overall,
    ( SELECT english_requirements.overall
           FROM public.english_requirements
          WHERE ((english_requirements.course_id = c.id) AND (english_requirements.test_type = 'TOEFL'::text))
         LIMIT 1) AS toefl_overall,
    ( SELECT english_requirements.overall
           FROM public.english_requirements
          WHERE ((english_requirements.course_id = c.id) AND (english_requirements.test_type = ANY (ARRAY['CAE'::text, 'Cambridge'::text])))
         LIMIT 1) AS cae_overall,
    ( SELECT english_requirements.overall
           FROM public.english_requirements
          WHERE ((english_requirements.course_id = c.id) AND (english_requirements.test_type = ANY (ARRAY['Duolingo'::text, 'DET'::text])))
         LIMIT 1) AS duolingo_overall,
        CASE
            WHEN (c.duration IS NULL) THEN NULL::double precision
            WHEN (c.duration_term ~~* 'month%'::text) THEN ((c.duration)::double precision / (12.0)::double precision)
            WHEN (c.duration_term ~~* 'week%'::text) THEN ((c.duration)::double precision / (52.0)::double precision)
            ELSE (c.duration)::double precision
        END AS duration_years
   FROM (public.courses c
     JOIN public.universities u ON ((u.id = c.university_id)))
  WHERE ((COALESCE(c.status, 'active'::text) = 'active'::text) AND (COALESCE(c.approval_status, 'approved'::text) = 'approved'::text))
  WITH NO DATA;
