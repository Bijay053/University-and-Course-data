PROBLEM 2: Gemini Vision API intermittently returns 503 "UNAVAILABLE — high demand" errors. The code treats this as "no data" and gives up silently.
Confirmed: direct curl with the same image and same key works one minute, fails with 503 the next. The gemini-2.5-flash model is overloaded periodically. The code currently has zero retry logic.
Fix needed: Add retry-with-backoff to ALL Gemini API calls. Find every fetch call to generativelanguage.googleapis.com (search across scrape.ts, english-requirements.ts, browser-helper.ts, and the worker file) and replace with a wrapper:
typescriptasync function callGeminiWithRetry(
  url: string,
  body: any,
  maxRetries = 4,
): Promise<any> {
  const RETRY_STATUSES = [429, 500, 502, 503, 504];
  const BACKOFFS_MS = [2000, 5000, 10000, 20000];
  const FALLBACK_MODELS = ["gemini-2.5-flash", "gemini-flash-latest", "gemini-2.5-pro"];
  
  let lastError: any;
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    // Cycle through fallback models on retries
    const modelIdx = Math.min(attempt, FALLBACK_MODELS.length - 1);
    const currentUrl = url.replace(/models\/[^:]+:/, `models/${FALLBACK_MODELS[modelIdx]}:`);
    
    try {
      const res = await fetch(currentUrl, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(body),
        signal: AbortSignal.timeout(60_000),
      });
      
      if (res.ok) return await res.json();
      
      if (RETRY_STATUSES.includes(res.status) && attempt < maxRetries) {
        const wait = BACKOFFS_MS[attempt];
        console.log(`[gemini retry] HTTP ${res.status} on ${FALLBACK_MODELS[modelIdx]}, waiting ${wait}ms (attempt ${attempt + 1}/${maxRetries})`);
        await new Promise(r => setTimeout(r, wait));
        continue;
      }
      
      const errBody = await res.text();
      throw new Error(`Gemini API ${res.status} on ${FALLBACK_MODELS[modelIdx]}: ${errBody.slice(0, 200)}`);
    } catch (e: any) {
      lastError = e;
      if (attempt < maxRetries && (e.name === "TimeoutError" || e.code === "ECONNRESET" || e.code === "ETIMEDOUT")) {
        const wait = BACKOFFS_MS[attempt];
        console.log(`[gemini retry] network error: ${e.message}, waiting ${wait}ms`);
        await new Promise(r => setTimeout(r, wait));
        continue;
      }
      throw e;
    }
  }
  throw lastError;
}
