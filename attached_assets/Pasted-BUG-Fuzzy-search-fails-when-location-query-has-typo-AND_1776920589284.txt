Two bugs:
Bug A — Location doesn't use trigram fuzzy match when combined with course search.
The location query parameter must use the SAME trigram fuzzy matching the course name uses. Not exact/ILIKE.
sql-- Current likely uses ILIKE or equality
AND course_location ILIKE '%' || $location || '%'

-- Should use trigram % operator (same as course_name)
AND (
  LOWER(university_city) % LOWER($location) 
  OR LOWER(university_country) % LOWER($location)
  OR LOWER(COALESCE(course_location, '')) % LOWER($location)
)
Also lower similarity threshold for location (short words like "syd" need more tolerance):
sqlSET LOCAL pg_trgm.similarity_threshold = 0.2;  -- or use word_similarity()
Bug B — Informal location aliases not supported.
Users use colloquial terms. Add alias expansion BEFORE the trigram search:
typescriptconst LOCATION_ALIASES: Record<string, string[]> = {
  // Country aliases
  'aussie': ['australia'],
  'oz': ['australia'],
  'au': ['australia'],
  'usa': ['united states', 'america'],
  'uk': ['united kingdom', 'england', 'britain'],
  'britain': ['united kingdom'],
  'england': ['united kingdom'],
  
  // City aliases (AU focus)
  'syd': ['sydney'],
  'melb': ['melbourne'],
  'bris': ['brisbane'],
  'adel': ['adelaide'],
  'per': ['perth'],
  'canberra': ['canberra', 'act'],
  
  // Common typos handled via trigram already: sidney→sydney, melbone→melbourne
};

function expandLocationQuery(input: string): string[] {
  const key = input.toLowerCase().trim();
  const aliases = LOCATION_ALIASES[key];
  return aliases ? [...aliases, input] : [input];
}

// In the query builder:
