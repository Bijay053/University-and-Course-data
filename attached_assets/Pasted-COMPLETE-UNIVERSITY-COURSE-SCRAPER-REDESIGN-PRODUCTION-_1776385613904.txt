ROOT CAUSE ANALYSIS
Why The Current Scraper Failed:

No Research Phase - Blindly fetched sitemap URLs without validating if they were actual course pages
Wrong Model Usage - Used deprecated Gemini models (gemini-2.0-flash, gemini-2.5-flash-preview) causing API failures
Ignored Pagination - Stopped at page 1 when sites showed "612 courses" across 62 pages
No Dynamic Site Support - Failed on JavaScript-rendered course listings
Sitemap Over-Reliance - Scraped navigation links, accommodation pages, student life pages from sitemaps
No Multi-Page Data Extraction - Missed fees on separate fee pages, requirements on separate requirement pages
Wasteful AI Usage - Called AI for every page instead of using reusable extraction patterns
No Validation - Saved incomplete/wrong data directly to production without approval
Poor Progress Tracking - User couldn't see what was being scraped or stop it
No Staging Workflow - Replaced live data without review

class ScraperOrchestrator:

    async def start_scrape(
        self, 
        university_id: int,
        url: str,
        job_id: str
    ) -> Dict:
        """
        Main entry point for scraping
        
        Returns:
        {
            'status': 'success' | 'failed' | 'needs_approval',
            'job_id': str,
            'research_summary': dict,
            'courses_discovered': int,
            'courses_extracted': int,
            'estimated_cost': float,
            'confidence': float
        }
        """
        
        job = await ScrapeJob.get(job_id)
        university = await University.get(university_id)
        
        try:
            # ═══════════════════════════════════════════════════════
            # PHASE 1: URL ANALYSIS
            # ═══════════════════════════════════════════════════════
            await self._update_job_status(job, 'analyzing_url', {
                'message': 'Analyzing URL type and site structure...'
            })
            
            url_analysis = await self.url_analyzer.analyze(url, university.country)
            
            # url_analysis = {
            #     'url_type': 'home' | 'course_listing' | 'sitemap' | 'single_course',
            #     'is_dynamic': bool,
            #     'suggested_listing_url': str | None,
            #     'pagination_detected': bool,
            #     'total_courses_hint': int | None
            # }
            
            # If home page, auto-discover course listing
            if url_analysis['url_type'] == 'home' and url_analysis['suggested_listing_url']:
                url = url_analysis['suggested_listing_url']
                await self._update_job_status(job, 'url_redirected', {
                    'message': f'Redirected to course listing: {url}'
                })
            
            # ═══════════════════════════════════════════════════════
            # PHASE 2: RESEARCH & VALIDATION
            # ═══════════════════════════════════════════════════════
            await self._update_job_status(job, 'researching', {
                'message': 'Sampling course pages for validation...'
            })
            
            # Get sample URLs (5-10 random samples)
            sample_urls = await self.course_discovery.get_sample_urls(
                url, 
                url_analysis,
                limit=5
            )
            
            # Validate samples
            validation_results = await self.research_validator.validate_samples(
                sample_urls,
                university.country
            )
            
            # validation_results = {
            #     'total_samples': 5,
            #     'valid_samples': 4,
            #     'invalid_samples': 1,
            #     'success_rate': 0.8,
            #     'valid_examples': [...],
            #     'invalid_examples': [...],
            #     'learned_patterns': {...}
            # }
            
            # Check if validation passed (need 80%+ success)
            if validation_results['success_rate'] < 0.8:
                await self._update_job_status(job, 'failed', {
                    'message': f'Low validation success rate: {validation_results["success_rate"]:.0%}',
                    'details': validation_results
                })
                return {
                    'status': 'failed',
                    'reason': 'validation_failed',
                    'details': validation_results
                }
            
            # ═══════════════════════════════════════════════════════
            # PHASE 3: COURSE DISCOVERY
            # ═══════════════════════════════════════════════════════
            await self._update_job_status(job, 'discovering_courses', {
                'message': 'Discovering all course URLs...'
            })
            
            # Get all course URLs using learned patterns
            all_course_urls = await self.course_discovery.discover_all_courses(
                url,
                url_analysis,
                validation_results['learned_patterns'],
                use_browser=url_analysis['is_dynamic']
            )
            
            # Filter using validation patterns
            filtered_urls = self.course_discovery.filter_valid_courses(
                all_course_urls,
                validation_results['learned_patterns']
            )
            
            estimated_cost = len(filtered_urls) * 0.01  # $0.01 per course estimate
            
            # Request approval before proceeding
            await self._update_job_status(job, 'awaiting_approval', {
                'message': 'Ready to scrape. Awaiting approval...',
                'research_summary': {
                    'url_type': url_analysis['url_type'],
                    'is_dynamic': url_analysis['is_dynamic'],
                    'samples_validated': validation_results['total_samples'],
                    'validation_success_rate': validation_results['success_rate'],
                    'valid_examples': validation_results['valid_examples'][:3],
                    'invalid_examples': validation_results['invalid_examples'][:3],
                    'total_courses_discovered': len(filtered_urls),
                    'estimated_cost_usd': round(estimated_cost, 2),
                    'estimated_time_minutes': len(filtered_urls) * 0.5  # 30 sec per course
                }
            })
            
            return {
                'status': 'needs_approval',
                'job_id': job_id,
                'research_summary': job.progress_data['research_summary']
            }
            
        except Exception as e:
            await self._update_job_status(job, 'failed', {
                'message': f'Error: {str(e)}',
                'error': str(e)
            })
            raise
    
    
    async def continue_after_approval(self, job_id: str) -> Dict:
        """
        Continue scraping after user approval
        """
        
        job = await ScrapeJob.get(job_id)
        university = await University.get(job.university_id)
        
        # Get filtered URLs from research phase
        filtered_urls = job.progress_data.get('filtered_urls', [])
        
        # Check for saved extraction patterns
        saved_patterns = await self.pattern_storage.get_patterns(university.id)
        
        extracted_courses = []
        errors = []
        
        # ═══════════════════════════════════════════════════════
        # PHASE 4: MULTI-PAGE DATA EXTRACTION
        # ═══════════════════════════════════════════════════════
        
        for idx, course_url in enumerate(filtered_urls):
            try:
                await self._update_job_status(job, 'extracting', {
                    'message': f'Extracting course {idx + 1} of {len(filtered_urls)}',
                    'current_url': course_url,
                    'progress': (idx + 1) / len(filtered_urls)
                })
                
                # Extract course data with multi-page support
                course_data = await self._extract_complete_course(
                    course_url,
                    university,
                    saved_patterns
                )
                
                extracted_courses.append(course_data)
                
                # Small delay to avoid rate limiting
                await asyncio.sleep(0.5)
                
            except Exception as e:
                errors.append({
                    'url': course_url,
                    'error': str(e)
                })
                continue
        
        # ═══════════════════════════════════════════════════════
        # PHASE 5: VALIDATION & STAGING
        # ═══════════════════════════════════════════════════════
        
        await self._update_job_status(job, 'validating', {
            'message': 'Validating extracted data...'
        })
        
        validated_courses = []
        for course in extracted_courses:
            validation = self.validator.validate_course(course)
            course['_validation'] = validation
            validated_courses.append(course)
        
        # Save to staging
        staged_ids = []
        for course in validated_courses:
            staged = await StagedCourse.create(
                job_id=job_id,
                university_id=university.id,
                data=course,
                validation_score=course['_validation']['score'],
                missing_fields=course['_validation']['missing_fields']
            )
            staged_ids.append(staged.id)
        
        # ═══════════════════════════════════════════════════════
        # PHASE 6: SAVE REUSABLE PATTERNS
        # ═══════════════════════════════════════════════════════
        
        if len(validated_courses) > 0:
            # Learn and save successful extraction patterns
            learned_patterns = self._learn_extraction_patterns(
                validated_courses,
                filtered_urls[:10]  # Use first 10 for pattern learning
            )
            
            await self.pattern_storage.save_patterns(
                university.id,
                learned_patterns
            )
        
        await self._update_job_status(job, 'completed', {
            'message': 'Scraping completed. Data staged for approval.',
            'courses_extracted': len(validated_courses),
            'courses_with_errors': len(errors),
            'staged_course_ids': staged_ids
        })
        
        return {
            'status': 'success',
            'job_id': job_id,
            'courses_extracted': len(validated_courses),
            'courses_staged': len(staged_ids),
            'errors': errors[:10]  # First 10 errors only
        }
    
    
    async def _extract_complete_course(
        self,
        course_url: str,
        university: University,
        saved_patterns: Optional[Dict] = None
    ) -> Dict:
        """
        Extract complete course data with multi-page support
        
        Steps:
        1. Fetch main course page
        2. Extract basic info (name, description, degree level)
        3. Discover linked pages (fees, requirements, scholarships)
        4. Fetch and extract from each linked page
        5. Merge all data into complete record
        """
        
        # Fetch main course page
        if saved_patterns and saved_patterns.get('is_dynamic'):
            html = await self.browser_handler.fetch(course_url)
        else:
            html = await self._fetch_html(course_url)
        
        # Initialize course data
        course_data = {
            'course_url': course_url,
            'university_name': university.name
        }
        
        # ─────────────────────────────────────────────────────
        # Extract from main course page
        # ─────────────────────────────────────────────────────
        
        # Try saved patterns first (rule-based, fast, free)
        if saved_patterns:
            course_data.update(
                await self._extract_with_patterns(html, saved_patterns)
            )
        else:
            # No saved patterns - use extractors
            for name, extractor in self.extractors.items():
                try:
                    result = await extractor.extract(html, course_url, university.country)
                    course_data.update(result)
                except Exception as e:
                    print(f"Extractor {name} failed: {e}")
        
        # ─────────────────────────────────────────────────────
        # Multi-page extraction
        # ─────────────────────────────────────────────────────
        
        # Discover linked pages
        linked_pages = self._discover_linked_pages(html, course_url)
        
        # linked_pages = {
        #     'fee_page': 'https://...',
        #     'requirements_page': 'https://...',
        #     'scholarship_page': 'https://...'
        # }
        
        # Extract fees from dedicated fee page if available
        if linked_pages.get('fee_page') and not course_data.get('international_fee'):
            fee_html = await self._fetch_html(linked_pages['fee_page'])
            fee_data = await self.extractors['fee'].extract(
                fee_html, 
                linked_pages['fee_page'],
                university.country
            )
            course_data.update(fee_data)
        
        # Extract requirements from dedicated page
        if linked_pages.get('requirements_page'):
            req_html = await self._fetch_html(linked_pages['requirements_page'])
            
            # English tests
            if not course_data.get('ielts_overall'):
                english_data = await self.extractors['english'].extract(
                    req_html,
                    linked_pages['requirements_page'],
                    university.country
                )
                course_data.update(english_data)
            
            # Academic requirements
            if not course_data.get('academic_requirements'):
                academic_data = await self.extractors['academic'].extract(
                    req_html,
                    linked_pages['requirements_page'],
                    university.country
                )
                course_data.update(academic_data)
        
        # Extract scholarships from dedicated page
        if linked_pages.get('scholarship_page'):
            scholarship_html = await self._fetch_html(linked_pages['scholarship_page'])
            scholarship_data = await self.extractors['scholarship'].extract(
                scholarship_html,
                linked_pages['scholarship_page'],
                university.country
            )
            course_data.update(scholarship_data)
        
        # ─────────────────────────────────────────────────────
        # AI fallback for missing critical fields
        # ─────────────────────────────────────────────────────
        
        critical_missing = self._identify_critical_missing(course_data)
        
        if critical_missing:
            # Use AI only for missing critical fields
            ai_extracted = await self.ai_fallback.extract_missing_fields(
                html,
                critical_missing,
                university.country
            )
            course_data.update(ai_extracted)
        
        return course_data
    
    
    def _discover_linked_pages(self, html: str, base_url: str) -> Dict[str, str]:
        """
        Discover linked pages for fees, requirements, scholarships
        """
        from bs4 import BeautifulSoup
        from urllib.parse import urljoin
        import re
        
        soup = BeautifulSoup(html, 'html.parser')
        linked_pages = {}
        
        # Find fee page links
        fee_patterns = [
            r'fees?',
            r'tuition',
            r'costs?',
            r'pricing'
        ]
        
        for link in soup.find_all('a', href=True):
            href = link.get('href')
            text = link.get_text(strip=True).lower()
            
            # Fee page
            if any(re.search(pattern, text, re.I) or re.search(pattern, href, re.I) 
                   for pattern in fee_patterns):
                if 'fee_page' not in linked_pages:
                    linked_pages['fee_page'] = urljoin(base_url, href)
            
            # Requirements page
            if any(kw in text or kw in href.lower() for kw in [
                'requirement', 'entry', 'admission', 'apply', 'eligibility'
            ]):
                if 'requirements_page' not in linked_pages:
                    linked_pages['requirements_page'] = urljoin(base_url, href)
            
            # Scholarship page
            if any(kw in text or kw in href.lower() for kw in [
                'scholarship', 'bursary', 'financial aid'
            ]):
                if 'scholarship_page' not in linked_pages:
                    linked_pages['scholarship_page'] = urljoin(base_url, href)
        
        return linked_pages
    
    
    def _identify_critical_missing(self, course_data: Dict) -> List[str]:
        """
        Identify critical missing fields that need AI extraction
        """
        critical_fields = [
            'course_name',
            'degree_level',
            'duration',
            'international_fee',
            'study_mode'
        ]
        
        missing = []
        for field in critical_fields:
            if not course_data.get(field):
                missing.append(field)
        
        return missing
    
    
    async def _update_job_status(self, job: ScrapeJob, status: str, data: Dict):
        """Update job status and progress"""
        job.status = status
        job.progress_data = {**job.progress_data, **data}
        job.updated_at = datetime.utcnow()
        await job.save()
    
    
    async def _fetch_html(self, url: str) -> str:
        """Simple HTTP fetch"""
        import aiohttp
        async with aiohttp.ClientSession() as session:
            async with session.get(url, timeout=30) as response:
                return await response.text()

pythonimport re
from typing import Dict, Optional
from .base_extractor import BaseExtractor


class FeeExtractor(BaseExtractor):
    """
    Advanced fee discovery with multi-page support
    """
    
    def __init__(self):
        self.currency_map = {
            'Australia': 'AUD',
            'New Zealand': 'NZD',
            'Canada': 'CAD',
            'USA': 'USD',
            'United States': 'USD',
            'United Kingdom': 'GBP',
            'UK': 'GBP',
            'Singapore': 'SGD',
            'Ireland': 'EUR'
        }
    
    async def extract(self, html: str, url: str, country: str) -> Dict:
        """
        Extract international fee, fee term, fee year, currency
        """
        
        # Auto-detect currency by country
        currency = self.currency_map.get(country, 'USD')
        
        # ─────────────────────────────────────────────────────
        # Extract all fee amounts from page
        # ─────────────────────────────────────────────────────
        
        all_fees = self._extract_all_fee_amounts(html)
        
        # ─────────────────────────────────────────────────────
        # Identify international vs domestic fees
        # ─────────────────────────────────────────────────────
        
        international_fee = self._identify_international_fee(html, all_fees)
        
        # ─────────────────────────────────────────────────────
        # Extract fee term (semester/annual/full course)
        # ─────────────────────────────────────────────────────
        
        fee_term = self._extract_fee_term(html)
        
        # ─────────────────────────────────────────────────────
        # Extract fee year
        # ─────────────────────────────────────────────────────
        
        fee_year = self._extract_fee_year(html)
        
        return {
            'international_fee': international_fee,
            'fee_term': fee_term,
            'fee_year': fee_year,
            'currency': currency
        }
    
    
    def _extract_all_fee_amounts(self, html: str) -> list:
        """
        Extract all fee amounts from HTML
        """
        # Remove HTML tags for cleaner text matching
        text = re.sub(r'<[^>]+>', ' ', html)
        
        # Patterns for fee amounts
        patterns = [
            r'(?:AUD|NZD|CAD|USD|GBP|€|\$|£)\s*([\d,]+)',  # $25,000
            r'([\d,]+)\s*(?:AUD|NZD|CAD|USD|GBP)',  # 25,000 AUD
        ]
        
        amounts = []
        for pattern in patterns:
            matches = re.findall(pattern, text)
            for match in matches:
                # Clean and convert to float
                amount = float(match.replace(',', ''))
                if 5000 <= amount <= 100000:  # Reasonable fee range
                    amounts.append(amount)
        
        return amounts
    
    
    def _identify_international_fee(self, html: str, all_fees: list) -> Optional[float]:
        """
        Identify international fee from all extracted fees
        
        Logic:
        1. Look for explicit "international" label
        2. If both domestic and international present, choose higher amount
        3. Filter reasonable range
        """
        
        text = re.sub(r'<[^>]+>', ' ', html).lower()
        
        # Pattern 1: Explicit international fee
        intl_patterns = [
            r'international\s+(?:student|fee|tuition)[\s:]*(?:aud|nzd|cad|usd|gbp|€|\$|£)?\s*([\d,]+)',
            r'overseas\s+(?:student|fee|tuition)[\s:]*(?:aud|nzd|cad|usd|gbp|€|\$|£)?\s*([\d,]+)',
        ]
        
        for pattern in intl_patterns:
            match = re.search(pattern, text, re.I)
            if match:
                return float(match.group(1).replace(',', ''))
        
        # Pattern 2: Table with domestic and international columns
        # (international fees are usually higher)
        if len(all_fees) >= 2:
            # Assume higher amount is international
            return max(all_fees)
        
        # Pattern 3: Single fee found
        if len(all_fees) == 1:
            return all_fees[0]
        
        return None
    
    
    def _extract_fee_term(self, html: str) -> str:
        """
        Extract fee term: Semester, Annual, Trimester, Full Course
        """
        
        text = re.sub(r'<[^>]+>', ' ', html).lower()
        
        fee_term_patterns = {
            'Semester': r'per\s+semester|\/\s*semester|each\s+semester',
            'Annual': r'per\s+year|per\s+annum|annually|\/\s*year|yearly',
            'Trimester': r'per\s+trimester|\/\s*trimester',
            'Full Course': r'total\s+course|full\s+course|entire\s+program|complete\s+course'
        }
        
        for term, pattern in fee_term_patterns.items():
            if re.search(pattern, text, re.I):
                return term
        
        # Default to Annual if not specified
        return 'Annual'
    
    
    def _extract_fee_year(self, html: str) -> Optional[int]:
        """
        Extract fee year (2025, 2026, etc.)
        """
        
        text = re.sub(r'<[^>]+>', ' ', html)
        
        # Pattern: "2026 fees" or "fees for 2025"
        pattern = r'(?:202[4-9]|20[3-9]\d)\s+fee|fee[s]?\s+for\s+(202[4-9]|20[3-9]\d)'
        match = re.search(pattern, text, re.I)
        
        if match:
            year = re.search(r'202[4-9]|20[3-9]\d', match.group(0))
            if year:
                return int(year.group(0))
        
        return None

    def _extract_ielts(self, text: str) -> Optional[Dict]:
        """
        Extract IELTS with band logic
        
        Patterns:
        - "IELTS Overall 6.5 with no band less than 6.0"
        - "IELTS 6.5 (6.0 in each band)"
        - "IELTS 7.0 (L:6.5, R:6.5, W:7.0, S:7.0)"
        """
        
        # Pattern 1: Overall with minimum band
        pattern1 = r'IELTS.*?(?:overall|minimum|total)[:\s]*([\d.]+).*?(?:no\s+(?:band|score)\s+(?:less|lower)\s+than|minimum\s+of)[:\s]*([\d.]+)'
        match = re.search(pattern1, text, re.I)
        if match:
            overall = float(match.group(1))
            min_band = float(match.group(2))
            return {
                'overall': overall,
                'listening': min_band,
                'reading': min_band,
                'writing': min_band,
                'speaking': min_band
            }
        
        # Pattern 2: Overall with each band
        pattern2 = r'IELTS[:\s]*([\d.]+)\s*\([\s]*([\d.]+)\s+(?:in\s+)?each'
        match = re.search(pattern2, text, re.I)
        if match:
            overall = float(match.group(1))
            each_band = float(match.group(2))
            return {
                'overall': overall,
                'listening': each_band,
                'reading': each_band,
                'writing': each_band,
                'speaking': each_band
            }
        
        # Pattern 3: Detailed breakdown
        pattern3 = r'IELTS[:\s]*([\d.]+).*?L(?:istening)?[:\s]*([\d.]+).*?R(?:eading)?[:\s]*([\d.]+).*?W(?:riting)?[:\s]*([\d.]+).*?S(?:peaking)?[:\s]*([\d.]+)'
        match = re.search(pattern3, text, re.I)
        if match:
            return {
                'overall': float(match.group(1)),
                'listening': float(match.group(2)),
                'reading': float(match.group(3)),
                'writing': float(match.group(4)),
                'speaking': float(match.group(5))
            }
        
        # Pattern 4: Simple overall only
        pattern4 = r'IELTS[:\s]*([\d.]+)'
        match = re.search(pattern4, text, re.I)
        if match:
            return {
                'overall': float(match.group(1))
            }
        
        return None
