pythonENGLISH_SECTION_HEADING_RE = re.compile(
    r"(English\s+(?:Language\s+)?Requirements?|Entry\s+Requirements?)",
    re.IGNORECASE,
)

def _find_english_section_images(html: str, base_url: str) -> list[str]:
    """
    Find images inside a section whose heading matches English/Entry requirements.
    Filename-agnostic — works regardless of how images are named.
    """
    soup = BeautifulSoup(html, "lxml")
    images = []
    for text_node in soup.find_all(string=ENGLISH_SECTION_HEADING_RE):
        # walk up to a reasonable container (section/article/div)
        container = text_node.find_parent(["section", "article", "div"])
        if not container:
            continue
        # then take any image within that container
        for img in container.find_all("img"):
            src = img.get("src")
            if src:
                full_url = urljoin(base_url, src)
                if full_url not in images:
                    images.append(full_url)
    return images
Use this BEFORE the existing keyword/dimension filters. Images found via this method get top priority — they're definitionally candidates because they're inside the English requirements section of the page.
Validate OCR results contain English-test keywords
For any image OCR'd, verify the resulting text contains at least one of IELTS|PTE|TOEFL|Cambridge|Duolingo. If not, discard the result and don't write evidence — it's likely a false positive (an image of something else inside the English section, like a logo or a "How to apply" diagram).
pythondef _is_valid_english_ocr_result(ocr_text: str) -> bool:
    if not ocr_text:
