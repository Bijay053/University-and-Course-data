Subject: Priority 5 — Per-field fill-rate tracking + alerting
Build observability so we catch data quality regressions automatically instead of manually spot-checking the Evidence Review panel. The bugs we've debugged in recent weeks (vision hallucinating from logos, sibling cache propagating wrong values, AI fallback fabricating intakes, fee_term flips) all sat in staging undetected because we have no metrics infrastructure. This adds it.
This is a system, not a bug fix. Estimated 1-1.5 days. The four components are independent and can ship sequentially with verification at each step.

async def compute_run_metrics(scrape_run_id: int, university_id: int) -> None:
    """Compute per-field per-method metrics for a completed scrape run."""
    
    # Total successfully-staged courses for this run
    courses_total = await session.execute(
        select(func.count(ScrapedCourse.id))
        .where(ScrapedCourse.scrape_run_id == scrape_run_id)
        .where(ScrapedCourse.status.in_(['pending', 'review_ready']))
    ).scalar()
    
    if courses_total == 0:
        return  # nothing to measure
    
    # Per-field, per-method evidence counts
    rows = await session.execute(
        select(
            ScrapedFieldEvidence.field_key,
            ScrapedFieldEvidence.method,
            func.count(distinct(ScrapedFieldEvidence.scraped_course_id)).label('count')
        )
        .join(ScrapedCourse, ScrapedCourse.id == ScrapedFieldEvidence.scraped_course_id)
        .where(ScrapedCourse.scrape_run_id == scrape_run_id)
        .where(ScrapedFieldEvidence.selected == True)
        .group_by(ScrapedFieldEvidence.field_key, ScrapedFieldEvidence.method)
    ).all()
    
    metrics = []
    for row in rows:
        metrics.append(ScrapeRunMetrics(
            scrape_run_id=scrape_run_id,
            university_id=university_id,
            field_key=row.field_key,
            method=row.method,
            count=row.count,
            courses_total=courses_total,
            fill_rate=row.count / courses_total,
        ))
    
    session.add_all(metrics)
    await session.commit()

python# backend-py/scripts/seed_baselines.py

async def seed_baselines() -> None:
    """Populate university_field_baselines from last 30 days of metrics.
    
    Excludes runs that resulted in known-bad data (manually flagged) so
    the baseline reflects what 'good' looks like.
    """
    cutoff = datetime.now() - timedelta(days=30)
    
    # Get the median fill rate per university per field across recent runs
    median_rates = await session.execute("""
        WITH recent_runs AS (
            SELECT 
                m.university_id,
                m.field_key,
                m.scrape_run_id,
                SUM(m.fill_rate) as total_fill_rate,
                jsonb_object_agg(m.method, m.fill_rate) as method_dist
            FROM scrape_run_metrics m
            JOIN scrape_runtime_jobs j ON j.id = m.scrape_run_id
            WHERE j.completed_at > :cutoff
              AND j.status = 'completed'
              AND COALESCE(j.flagged_bad, false) = false
            GROUP BY m.university_id, m.field_key, m.scrape_run_id
        )
        SELECT 
            university_id,
            field_key,
            percentile_cont(0.5) WITHIN GROUP (ORDER BY total_fill_rate) as median_fill_rate,
            percentile_cont(0.10) WITHIN GROUP (ORDER BY total_fill_rate) as p10_fill_rate,
            COUNT(*) as sample_size
        FROM recent_runs
        GROUP BY university_id, field_key
        HAVING COUNT(*) >= 3  -- need at least 3 runs to baseline
    """, {"cutoff": cutoff})
    
    for row in median_rates:
        baseline = UniversityFieldBaseline(
            university_id=row.university_id,
            field_key=row.field_key,
            expected_fill_rate=row.median_fill_rate,
            floor_threshold=max(0.50, row.p10_fill_rate - 0.05),  # p10 minus 5pp, but never below 50%
            expected_method_distribution=...,  # similar median calculation
            sample_size=row.sample_size,
        )
        await session.merge(baseline)
    
    await session.commit()

    "international_fee": 0.70,
    "ielts_overall": 0.85,
    "duration": 0.90,
    "intake_months": 0.85,
    "course_name": 0.99,  # course name should always populate
    # ... all tracked fields
}

# Method distribution rules — flag if these methods exceed thresholds
METHOD_QUALITY_RULES = [
    {
        "rule_id": "vision_dominance",
        "description": "Vision OCR producing >40% of values for a field suggests "
                       "page-text extractors are failing or vision is hallucinating",
        "fields": ["ielts_overall", "toefl_overall", "duolingo_overall", "cambridge_overall"],
        "method": "per_course_vision",
        "threshold": 0.40,
        "severity": "warning",

        "description": "Sibling cache producing >30% of values means most values "
                       "are propagated, not extracted per-course",
        "fields": "*",
        "method_pattern": "sibling_cache:*",
        "threshold": 0.30,
        "severity": "warning",
    },
    {
        "rule_id": "ai_fallback_dominance",
        "description": "AI fallback producing >25% of values means primary "
                       "extractors are failing",
        "fields": "*",
        "method": "ai_fallback",
        "threshold": 0.25,
        "severity": "critical",
    },
    {
        "rule_id": "facebook_icon_source",
        "description": "Evidence source URL contains 'facebook' or 'instagram' or "
                       "'icon' — should never happen for English fields",
        "fields": ["ielts_overall", "toefl_overall", "pte_overall", 
                   "duolingo_overall", "cambridge_overall"],
        "url_pattern": r"(facebook|instagram|icon[-_]|^link\.|logo)",
        "threshold": 0.0,  # any occurrence is wrong
        "severity": "critical",
    },
]


async def evaluate_run_alerts(scrape_run_id: int) -> list[Alert]:
    """Check a completed run against baselines + global rules."""
    alerts = []
    
    # Load this run's metrics
    metrics = await load_run_metrics(scrape_run_id)
    if not metrics:
        return []
    
    university_id = metrics[0].university_id
    
    # Load baselines for this university
    baselines = await load_baselines(university_id)
    
    # Check 1: per-field fill-rate vs baseline floor
    field_totals = aggregate_by_field(metrics)
    for field_key, total_rate in field_totals.items():
        baseline = baselines.get(field_key)
        floor = baseline.floor_threshold if baseline else GLOBAL_DEFAULT_FLOORS.get(field_key, 0.50)
        
        if total_rate < floor:
            severity = "critical" if total_rate < (floor - 0.20) else "warning"
            alerts.append(Alert(
                scrape_run_id=scrape_run_id,
                rule_id=f"fill_rate_drop:{field_key}",
                severity=severity,
                message=f"{field_key} fill-rate {total_rate:.0%} below floor {floor:.0%}",
                expected=floor,
                actual=total_rate,
            ))
    
    # Check 2: method-distribution rules
    for rule in METHOD_QUALITY_RULES:
        violations = check_method_rule(metrics, rule)
        for v in violations:
            alerts.append(Alert(
                scrape_run_id=scrape_run_id,
                rule_id=rule["rule_id"],
                severity=rule["severity"],
                message=f"{rule['description']} (field={v.field}, "
                        f"method={v.method}, rate={v.actual:.0%})",
                expected=rule.get("threshold"),
                actual=v.actual,
            ))
    
    # Check 3: trend detection vs last 5 runs
    recent_runs = await get_last_n_runs(university_id, n=5, exclude=scrape_run_id)
    if len(recent_runs) >= 3:
        for field_key, current_rate in field_totals.items():
            trend = compute_trend(recent_runs, field_key)
            if trend.is_significant_drop(current_rate):
                alerts.append(Alert(
                    scrape_run_id=scrape_run_id,
                    rule_id=f"trend_drop:{field_key}",
                    severity="warning",
                    message=f"{field_key} dropped {trend.drop_pct:.0%} below "
                            f"trailing 5-run average ({trend.mean:.0%})",
                ))
    
    return alerts


def check_method_rule(metrics: list, rule: dict) -> list:
    """Apply a single method-distribution rule to the metrics."""
    violations = []
    
    fields_to_check = (
        [m.field_key for m in metrics]  # all fields
        if rule["fields"] == "*"
        else rule["fields"]
    )
    
    for field in set(fields_to_check):
        field_metrics = [m for m in metrics if m.field_key == field]
        if not field_metrics:
            continue
        total = sum(m.count for m in field_metrics)
        if total == 0:
            continue
        
        # Match method or method pattern
        matching = [m for m in field_metrics 
                    if method_matches(m.method, rule)]
        matching_count = sum(m.count for m in matching)
        rate = matching_count / total
        
        if "url_pattern" in rule:
            # Need to check evidence URLs, not just counts
            bad_url_count = await count_evidence_with_url_pattern(
                field, rule["url_pattern"], metrics[0].scrape_run_id
            )
            rate = bad_url_count / total
        
        if rate > rule["threshold"]:
            violations.append(SimpleNamespace(
                field=field,
                method=rule.get("method", rule.get("method_pattern", "*")),
                actual=rate,
            ))
    
    return violations

sql-- View 1: Latest run health for each university
CREATE VIEW v_university_run_health AS
SELECT 
    u.code,
    u.name,
    j.id as scrape_run_id,
    j.completed_at,
    COUNT(a.id) FILTER (WHERE a.severity = 'critical') as critical_alerts,
    COUNT(a.id) FILTER (WHERE a.severity = 'warning') as warning_alerts,
    array_agg(DISTINCT a.rule_id) FILTER (WHERE a.severity = 'critical') as critical_rule_ids
FROM universities u
JOIN scrape_runtime_jobs j ON j.university_id = u.id
LEFT JOIN scrape_run_alerts a ON a.scrape_run_id = j.id
WHERE j.completed_at = (
    SELECT MAX(j2.completed_at) 
    FROM scrape_runtime_jobs j2 
    WHERE j2.university_id = u.id AND j2.status = 'completed'
)
GROUP BY u.code, u.name, j.id, j.completed_at
ORDER BY critical_alerts DESC, warning_alerts DESC;

-- View 2: Top deteriorating fields across all universities
CREATE VIEW v_field_health_trend AS
SELECT 
    field_key,
    AVG(fill_rate) FILTER (WHERE rn <= 5) as recent_avg,
    AVG(fill_rate) FILTER (WHERE rn BETWEEN 6 AND 10) as previous_avg,
    AVG(fill_rate) FILTER (WHERE rn <= 5) - 
        AVG(fill_rate) FILTER (WHERE rn BETWEEN 6 AND 10) as delta
FROM (
    SELECT 
        m.*,
        ROW_NUMBER() OVER (
            PARTITION BY m.university_id, m.field_key 
            ORDER BY m.computed_at DESC
        ) as rn
    FROM scrape_run_metrics m
) ranked
WHERE rn <= 10
GROUP BY field_key
HAVING AVG(fill_rate) FILTER (WHERE rn BETWEEN 6 AND 10) IS NOT NULL
ORDER BY delta ASC;

async def deliver_alerts(alerts: list[Alert]) -> None:
    """Send critical alerts to email/Slack. Warnings stay on dashboard only."""
    critical = [a for a in alerts if a.severity == "critical"]
    if not critical:
        return
    
    # Group by run for digest format
    by_run = group_by(critical, key=lambda a: a.scrape_run_id)
    
    for run_id, run_alerts in by_run.items():
        message = format_alert_digest(run_id, run_alerts)
        
        if SLACK_WEBHOOK_URL:
            await send_slack(SLACK_WEBHOOK_URL, message)
        
        if ALERT_EMAIL_TO:
            await send_email(
                to=ALERT_EMAIL_TO,
                subject=f"[Scraper alert] {len(run_alerts)} critical issues in run {run_id}",
                body=message,
            )


def format_alert_digest(run_id: int, alerts: list) -> str:
    lines = [f"Scrape run {run_id} produced {len(alerts)} critical alerts:\n"]
    for a in alerts:
        lines.append(f"• {a.rule_id}: {a.message}")
    lines.append(f"\nReview at: https://your-dashboard/runs/{run_id}")
    return "\n".join(lines)
Set SLACK_WEBHOOK_URL and ALERT_EMAIL_TO as env variables. If both are unset, the dashboard is the only output (which is fine).
