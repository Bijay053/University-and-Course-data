```typescript
interface AcademicRequirement {

function determineScoreType(scoreText: string): 'GPA' | 'Percentage' | 'CGPA' | 'Grade' {

}
```

```typescript
interface Intake {
  month: string; // "February", "July", "September"
  day?: number; // Specific start day if mentioned
  year?: number; // 2025, 2026
}

async function discoverIntakes(courseHtml: string): Promise<Intake[]> {
  const intakes: Intake[] = [];
  
  // Month patterns
  const monthPatterns = [
    // "Intakes: February, July, October"
    /(?:intake|start|commencement|entry)[s]?[:\s]*([A-Z][a-z]+(?:,\s*[A-Z][a-z]+)*)/i,
    
    // "Available in: Feb, Jul, Oct"
    /available\s+in[:\s]*([A-Z][a-z]+(?:,\s*[A-Z][a-z]+)*)/i,
    
    // "Start dates: 15 Feb 2025, 20 Jul 2025"
    /start\s+date[s]?[:\s]*([\d]+\s+[A-Z][a-z]+\s+\d{4}(?:,\s*[\d]+\s+[A-Z][a-z]+\s+\d{4})*)/i,
    
    // Table format
    /<td[^>]*>(?:Intake|Start|Entry)<\/td>\s*<td[^>]*>([^<]+)<\/td>/i
  ];
  
  for (const pattern of monthPatterns) {
    const match = courseHtml.match(pattern);
    if (match) {
      const intakeText = match[1];
      
      // Parse individual intakes
      const intakeStrings = intakeText.split(/,\s*/);
      
      for (const intakeStr of intakeStrings) {
        // Check for full date format: "15 February 2025"
        const fullDateMatch = intakeStr.match(/(\d+)\s+([A-Z][a-z]+)\s+(\d{4})/);
        if (fullDateMatch) {
          intakes.push({
            day: parseInt(fullDateMatch[1]),
            month: normalizeMonth(fullDateMatch[2]),
            year: parseInt(fullDateMatch[3])
          });
          continue;
        }
        
        // Check for month + year: "February 2025"
        const monthYearMatch = intakeStr.match(/([A-Z][a-z]+)\s+(\d{4})/);
        if (monthYearMatch) {
          intakes.push({
            month: normalizeMonth(monthYearMatch[1]),
            year: parseInt(monthYearMatch[2])
          });
          continue;
        }
        
        // Just month: "February"
        const monthMatch = intakeStr.match(/([A-Z][a-z]+)/);
        if (monthMatch) {
          intakes.push({
            month: normalizeMonth(monthMatch[1])
          });
        }
      }
      
      break; // Found intakes
    }
  }
  
  // Fallback: Search for semester patterns
  if (intakes.length === 0) {
    const semesterPatterns = {
      'Semester 1': ['February', 'March'],
      'Semester 2': ['July', 'August'],
      'Trimester 1': ['January', 'February'],
      'Trimester 2': ['May', 'June'],
      'Trimester 3': ['September', 'October']
    };
    
    for (const [semester, months] of Object.entries(semesterPatterns)) {
      if (new RegExp(semester, 'i').test(courseHtml)) {
        months.forEach(month => intakes.push({ month }));
      }
    }
  }
  
  return intakes;
}

function normalizeMonth(month: string): string {
  const monthMap: Record<string, string> = {
    'Jan': 'January', 'Feb': 'February', 'Mar': 'March',
    'Apr': 'April', 'May': 'May', 'Jun': 'June',
    'Jul': 'July', 'Aug': 'August', 'Sep': 'September',
    'Oct': 'October', 'Nov': 'November', 'Dec': 'December'
  };
  
  // Check if abbreviated
  for (const [abbr, full] of Object.entries(monthMap)) {
    if (month.toLowerCase().startsWith(abbr.toLowerCase())) {
      return full;
    }
  }
  
  return month;
}
```

```typescript
}

async function discoverCampuses(courseHtml: string, universityCountry: string): Promise<Campus[]> {
  const campuses: Campus[] = [];
  
  // Pattern 1: "Location: Sydney, Melbourne"
  const locationPattern = /(?:location|campus|city|cities|where)[s]?[:\s]*([A-Z][a-z]+(?:,\s*[A-Z][a-z]+)*)/i;
  
  // Pattern 2: "Available at: Main Campus (Sydney), City Campus (Melbourne)"
  const detailedPattern = /([A-Z][a-z\s]+Campus)\s*\(([^)]+)\)/gi;
  
  // Pattern 3: Table format
  const tablePattern = /<td[^>]*>(?:Location|Campus)<\/td>\s*<td[^>]*>([^<]+)<\/td>/i;
  
  // Try detailed pattern first
  let match;
  while ((match = detailedPattern.exec(courseHtml)) !== null) {
    const campusName = match[1].trim();
    const cityName = match[2].trim();
    
    campuses.push({
      campus: campusName,
      city: cityName,
      country: universityCountry
    });
  }
  
  // If no detailed matches, try simple location pattern
  if (campuses.length === 0) {
    const locationMatch = courseHtml.match(locationPattern);
    if (locationMatch) {
      const cities = locationMatch[1].split(/,\s*/);
      cities.forEach(city => {
        campuses.push({
          city: city.trim(),
          country: universityCountry
        });
      });
    }
  }
  
  // Multi-state detection (for Australia/USA/Canada)
  const statePatterns = {
    'Australia': /\b(NSW|VIC|QLD|SA|WA|TAS|NT|ACT)\b/g,
    'USA': /\b(AL|AK|AZ|AR|CA|CO|CT|DE|FL|GA|HI|ID|IL|IN|IA|KS|KY|LA|ME|MD|MA|MI|MN|MS|MO|MT|NE|NV|NH|NJ|NM|NY|NC|ND|OH|OK|OR|PA|RI|SC|SD|TN|TX|UT|VT|VA|WA|WV|WI|WY)\b/g,
    'Canada': /\b(AB|BC|MB|NB|NL|NT|NS|NU|ON|PE|QC|SK|YT)\b/g
  };
  
  const statePattern = statePatterns[universityCountry];
  if (statePattern) {
    const states = courseHtml.match(statePattern);
    if (states) {
      // Enhance existing campuses with state info
      campuses.forEach((campus, i) => {
        campus.state = states[i] || states[0];
      });
    }
  }
  
  return campuses;
}
```

### **PHASE 7: DURATION & STUDY MODE DISCOVERY**

```typescript
interface CourseDuration {
  duration: number;
  term: 'years' | 'months' | 'weeks' | 'semesters' | 'trimesters';
  studyMode: 'on_campus' | 'online' | 'blended' | 'both';
}

async function discoverDuration(courseHtml: string): Promise<CourseDuration> {
  
  // Duration patterns
  const durationPatterns = [
    // "3 years full-time"
    /([\d.]+)\s+(years?|months?|weeks?)/i,
    
    // "Duration: 2 years"
    /duration[:\s]*([\d.]+)\s+(years?|months?)/i,
    
    // "6 semesters"
    /([\d]+)\s+(semesters?|trimesters?)/i,
    
    // Table format
    /<td[^>]*>Duration<\/td>\s*<td[^>]*>([\d.]+)\s+([^<]+)<\/td>/i
  ];
  
  let duration = 0;
  let term: CourseDuration['term'] = 'years';
  
  for (const pattern of durationPatterns) {
    const match = courseHtml.match(pattern);
    if (match) {
      duration = parseFloat(match[1]);
      term = normalizeTerm(match[2]);
      break;
    }
  }
  
  // Study mode patterns
  const studyModePatterns = {
    'on_campus': /(?:on-campus|on campus|face-to-face|in-person)/i,
    'online': /(?:online|distance|remote|external)/i,
    'blended': /(?:blended|hybrid|mixed|both|online and on-campus)/i,
    'both': /(?:both|either|flexible)/i
  };
  
  let studyMode: CourseDuration['studyMode'] = 'on_campus'; // default
  
  for (const [mode, pattern] of Object.entries(studyModePatterns)) {
    if (pattern.test(courseHtml)) {
      studyMode = mode as CourseDuration['studyMode'];
      
      // Normalize "both" to "blended"
      if (studyMode === 'both') {
        studyMode = 'blended';
      }
      
      break;
    }
  }
  
  return { duration, term, studyMode };
}

function normalizeTerm(term: string): 'years' | 'months' | 'weeks' | 'semesters' | 'trimesters' {
  const normalized = term.toLowerCase().replace(/s$/, '') + 's';
  if (['years', 'months', 'weeks', 'semesters', 'trimesters'].includes(normalized)) {
    return normalized as any;
  }
  return 'years'; // default
}
```

```typescript
async function discoverScholarships(courseHtml: string, courseUrl: string, universityCurrency: string): Promise<Scholarship[]> {
  const scholarships: Scholarship[] = [];
  
  // Multi-page discovery
  const scholarshipSources = [
    { html: courseHtml, source: 'course_page' },
    { html: await fetchUrl(findScholarshipLink(courseUrl)), source: 'scholarship_page' },
    { html: await fetchUrl(findFinancialAidLink(courseUrl)), source: 'financial_aid_page' }
  ];
  
  for (const source of scholarshipSources) {
    const found = extractScholarships(source.html, universityCurrency);
    scholarships.push(...found);
  }
  
  return scholarships;
  
  function extractScholarships(html: string, currency: string): Scholarship[] {
    const results: Scholarship[] = [];
    
    // Pattern 1: "Merit Scholarship - Up to $10,000"
    const amountPattern = /([A-Z][^-:\n]+)\s*[-:]\s*(?:up to|worth|valued at)?\s*(?:\$|AUD|USD|GBP|CAD|NZD)?\s*([\d,]+)/gi;
    
    // Pattern 2: Table format
    const tablePattern = /<tr[^>]*>\s*<td[^>]*>([^<]+)<\/td>\s*<td[^>]*>([^<]+)<\/td>\s*<td[^>]*>([^<]+)<\/td>/gi;
    
    let match;
    while ((match = amountPattern.exec(html)) !== null) {
      const name = match[1].trim();
      const amount = parseFloat(match[2].replace(/,/g, ''));
      
      // Extract eligibility (look for text after scholarship name)
      const eligibilityPattern = new RegExp(`${name}[^.]*\\.\\s*([^.]+)`, 'i');
      const eligMatch = html.match(eligibilityPattern);
      const eligibility = eligMatch ? eligMatch[1].trim() : 'Based on merit';
      
      results.push({
        name,
        amount,
        currency,
        eligibility,
        details: `${name} - ${currency} ${amount}`
      });
    }
    
    return results;
  }
}
```

```typescript
function extractRelevantSection(html: string, field: string): string {
  const sectionKeywords = {
    'fees': ['fee', 'tuition', 'cost', 'price', 'payment'],
    'requirements': ['requirement', 'entry', 'admission', 'IELTS', 'TOEFL', 'PTE', 'academic'],
    'intakes': ['intake', 'start', 'commence', 'entry', 'semester', 'trimester'],
    'duration': ['duration', 'length', 'time', 'year', 'month'],
    'campus': ['location', 'campus', 'where', 'city', 'site']
  };
  
  const keywords = sectionKeywords[field] || [];
  
  // Find sections containing keywords
  const sections = [];
  const paragraphs = html.split(/<\/(?:p|div|section|table)>/i);
  
  for (const para of paragraphs) {
    if (keywords.some(kw => new RegExp(kw, 'i').test(para))) {
      sections.push(para);
    }
  }
  
  // Return relevant sections only (max 2000 chars)
  return sections.join('\n').slice(0, 2000);
}
```
