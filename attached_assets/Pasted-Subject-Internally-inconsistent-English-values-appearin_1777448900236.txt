pythondef validate_english_coherence(payload, log):
    """Flag implausible combinations of English values."""
    ielts = payload.get("ielts_overall")
    toefl = payload.get("toefl_overall")
    duolingo = payload.get("duolingo_overall")
    pte = payload.get("pte_overall")

    if not ielts:
        return  # nothing to anchor against

    expected = ENGLISH_EQUIVALENCE[ielts]  # {"toefl": (low, high), "duolingo": (low, high), ...}

    issues = []
    if toefl and not (expected["toefl"][0] - 5 <= toefl <= expected["toefl"][1] + 5):
        issues.append(f"TOEFL {toefl} inconsistent with IELTS {ielts}")
    if duolingo and not (expected["duolingo"][0] - 10 <= duolingo <= expected["duolingo"][1] + 10):
        issues.append(f"Duolingo {duolingo} inconsistent with IELTS {ielts}")
    if pte and not (expected["pte"][0] - 5 <= pte <= expected["pte"][1] + 5):
        issues.append(f"PTE {pte} inconsistent with IELTS {ielts}")

    if issues:
        # Flag for review, demote confidence on outlier fields
        for issue in issues:
            log.append({"level": "warning", "msg": f"English coherence: {issue}"})
