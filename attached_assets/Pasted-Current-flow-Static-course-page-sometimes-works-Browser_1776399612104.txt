Fix 3 — Improve browser extraction (CRITICAL)

Your browser fallback is weak.

Replace with this:

async function getBrowserRenderedText(page: any) {
  try {
    // wait for full render
    await page.waitForTimeout(1500);

    // click ALL possible requirement tabs
    const buttons = await page.$$("button, a");

    for (const btn of buttons) {
      const text = (await page.evaluate(el => el.innerText, btn))?.toLowerCase();

      if (
        text?.includes("requirement") ||
        text?.includes("entry") ||
        text?.includes("english") ||
        text?.includes("ielts")
      ) {
        try {
          await btn.click();
          await page.waitForTimeout(800);
        } catch {}
      }
    }

    // expand accordions
    const expanders = await page.$$('[aria-expanded="false"], button');

    for (const exp of expanders) {
      try {
        await exp.click();
      } catch {}
    }

    await page.waitForTimeout(1000);

    const text = await page.evaluate(() => document.body?.innerText || "");

    return text;
  } catch (e) {
    return "";
  }
}
