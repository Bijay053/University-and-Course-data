sql-- Find all duplicates within the same university where names differ only by case
-- Keep the OLDER course (lower id), merge populated fields from newer, delete newer

WITH dup_pairs AS (
  SELECT 
    MIN(id) as keep_id,
    MAX(id) as drop_id
  FROM courses
  WHERE status = 'active'
  GROUP BY university_id, LOWER(name)
  HAVING COUNT(*) = 2
)
UPDATE courses c
SET 
  course_location = COALESCE(NULLIF(c.course_location, ''), n.course_location),
  duration = COALESCE(c.duration, n.duration),
  duration_term = COALESCE(c.duration_term, n.duration_term),
  international_eligible = COALESCE(c.international_eligible, n.international_eligible),
  on_campus_available = COALESCE(c.on_campus_available, n.on_campus_available),
  eligibility_status = COALESCE(c.eligibility_status, n.eligibility_status),
  study_mode = COALESCE(c.study_mode, n.study_mode),
  description = COALESCE(c.description, n.description),
  updated_at = NOW()
FROM dup_pairs dp
JOIN courses n ON n.id = dp.drop_id
WHERE c.id = dp.keep_id;

typescriptimport multer from "multer";
const upload = multer({ storage: multer.memoryStorage() });

router.post("/universities/bulk-import", upload.single('csv'), async (req, res) => {
  if (!req.file) return res.status(400).json({ error: "CSV file required" });
  
  const csv = req.file.buffer.toString();
  const lines = csv.split('\n').filter(l => l.trim());
  if (lines.length < 2) return res.status(400).json({ error: "CSV needs header + at least 1 row" });
  
  const headers = lines[0].split(',').map(h => h.trim().toLowerCase());
  const required = ['name', 'website', 'country', 'city'];
  const missing = required.filter(r => !headers.includes(r));
  if (missing.length) return res.status(400).json({ error: `Missing columns: ${missing.join(', ')}` });
  
  const results = { created: 0, skipped: 0, errors: [] as string[] };
  
  for (let i = 1; i < lines.length; i++) {
    const cols = lines[i].split(',').map(c => c.trim());
    const row: Record<string, string> = {};
    headers.forEach((h, idx) => { row[h] = cols[idx] || ''; });
    
    if (!row.name || !row.website) {
      results.errors.push(`Row ${i+1}: missing name or website`);
      continue;
    }
    
    try {
      const existing = await findUniversityByNameCaseInsensitive(row.name);
      if (existing) { results.skipped++; continue; }
      
      await db.insert(universitiesTable).values({
        name: row.name,
        website: row.website,
        scrapeUrl: row.scrape_url || row.website,
        country: row.country || 'Unknown',
        city: row.city || 'Unknown',
      });
      results.created++;
    } catch (err) {
      results.errors.push(`Row ${i+1} (${row.name}): ${(err as Error).message}`);
    }
  }
  
  res.json(results);
});Frontend — new page artifacts/university-portal/src/pages/universities-bulk-import.tsx:
