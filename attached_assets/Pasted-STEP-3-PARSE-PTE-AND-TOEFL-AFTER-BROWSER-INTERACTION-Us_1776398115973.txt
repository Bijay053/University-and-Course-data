      writing: browserEnglish.pteWriting,
      speaking: browserEnglish.pteSpeaking,

  if (!toefl.overall && browserEnglish.toeflOverall) {

      overall: browserEnglish.toeflOverall,

if ((!pte.overall || !toefl.overall) && sharedRequirementsText) {
